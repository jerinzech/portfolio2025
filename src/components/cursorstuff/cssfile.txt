/* Custom Cursor Logic */
body {
  cursor: none; /* Hide default cursor */
}

/* We prefer a specific class for the cursor to handle the complex transforms */
.custom-cursor {
  /* Default CSS variables */
  --x: -100px;
  --y: -100px;
  --width: 24px;
  --height: 24px;
  --scale: 1;
  --translateX: 0;
  --translateY: 0;

  /* Positioning */
  position: fixed;
  left: 0;
  top: 0;
  transform: translate(calc(var(--x) - 50%), calc(var(--y) - 50%))
    scale(var(--scale));
  width: var(--width);
  height: var(--height);
  pointer-events: none; /* Crucial! Let clicks pass through */
  z-index: 9999;
  transition: width 0.3s ease, height 0.3s ease, transform 0.1s ease-out;
}

/* The inner dot styling */
.cursor-content {
  opacity: 0.3;
  transform: translate(var(--translateX), var(--translateY));
  transition: opacity 0.2s, transform 0.1s ease-out;
}

/* State: When locked onto a button/link */
.custom-cursor.is-locked .cursor-content {
  opacity: 0.1; /* Fades out slightly when locked */
}

/* Magnetic effect for links */
a,
button {
  transform: translate(var(--translateX, 0), var(--translateY, 0))
    scale(var(--scale, 1));
  transition: transform 0.1s ease-out;
}
